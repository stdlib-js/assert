    Tests if a value is a non-infinite positive number.

    Parameters
    ----------
    value: any
        Value to test.

    Returns
    -------
    bool: boolean
        Boolean indicating whether value is a 
        non-infinite positive number.

    Examples
    --------
    > var bool = {{alias}}( 5.0 )
    true
    > bool = {{alias}}( new Number( 5.0 ) )
    true
    > bool = {{alias}}( 3.14 )
    true
    > bool = {{alias}}( -5.0 )
    false
    > var bool = {{alias}}( 5.0/0.0 )
    false
    > bool = {{alias}}( new Number( 5.0/0.0 ) )
    false
    > bool = {{alias}}( null )
    false
